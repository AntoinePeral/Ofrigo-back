-----EXEMPLES DE REQUETTES-----
-------------------------------

-- AFFICHE TOUTES LES RECETTES ET LEURS TAGS
SELECT recipe.*, tag.label
FROM recipe
JOIN recipe_has_tag
ON recipe_id=recipe.id
JOIN tag
ON tag.id=tag_id

--AFFICHE UNE RECETTE ET CES TAGS
SELECT recipe.*, tag.label
FROM recipe
JOIN recipe_has_tag
ON recipe_id=recipe.id
JOIN tag
ON tag.id=tag_id
WHERE recipe.id=1

--AFFICHE LES TAGS DE LA RECETTE 2
SELECT tag.label
FROM tag
JOIN recipe_has_tag
ON tag_id=tag.id
WHERE recipe_id=2

-- AFFICHE TOUTES LES RECETTES ET LEURS INGREDIENTS
SELECT recipe.*, ingredient.label
FROM recipe
JOIN recipe_has_ingredient
ON recipe_id=recipe.id
JOIN ingredient
ON ingredient.id=ingredient_id

--AFFICHE LA RECETTE 1 ET CES INGREDIENT
SELECT recipe.*, ingredient.label
FROM recipe
JOIN recipe_has_ingredient
ON recipe_id=recipe.id
JOIN ingredient
ON ingredient.id=ingredient_id
WHERE recipe.id=1

--AFFICHE LES ETAPES DE LA RECETTE 3
SELECT step.content
FROM step
JOIN recipe
ON step.recipe_id=recipe.id
WHERE recipe.id=3

--AFFICHE TOUTES LES ETAPES DES RECETTES
SELECT step.content
FROM step
JOIN recipe
ON step.recipe_id=recipe.id

--AFFICHE L'UTILISATEUR ET SON MESSAGE
SELECT account.first_name, account.last_name, message.title, message.content
FROM account
JOIN message
ON account_id=account.id
WHERE account.id=1
